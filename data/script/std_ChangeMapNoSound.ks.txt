# std_ChangeMapNoSound.ks
# 指定されたパラメータに基づいて、フィールドマップ間の移動を行います。

PARAM={
	nextID # fieldMap.id
	x # int
	y #int 
	dir #FourDirection
	tooltip #I18NText
	
}


#------------------------------------------------



STEP_ON (FieldScriptAccessObject) ={
	#set tooltip
	setNode(nextID, x, y, dir, tooltip);
	
}

STEP_DOWN ( FieldScriptAccessObject ) ={
	#unset tooltip
	unsetNode();

}
