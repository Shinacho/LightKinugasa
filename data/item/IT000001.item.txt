# ----------------------------EQIP----------------------------

EQIP_REQUIRE( EqipRequireEvent ){
	set("STR").isOver("40");
	set("EQIP_ATTR").any();
	set("MAGIC").is("USABLE");

}

EQIP_EFFECT( EqipEffectEvent ){
	base("ATK").is("DCS_BASE_VALUE").add("STR").is("0.5f");
	base("DEF").is("8");
	addItemBagMax("4");
	addPassive("PassiveID");
}

ADDITIONAL_ACTION( ActionReloadEvent ){

		addAction("A000001");
	ENDIF

}
